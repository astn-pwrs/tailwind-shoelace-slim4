function isGroup($name) {
    return $redis->sIsMember("RBAC:GROUP:$name:USERS", $this->userId);
}

function hasGroupRole($group, $role) {
    $roleId = $this->roleNameToId($role);
    return $redis->sIsMember("RBAC:GROUP:$group:ROLES", $roleId);
}
